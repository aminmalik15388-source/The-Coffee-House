    <script>
        /**
         * The Coffee House - Complete JavaScript
         * Combined: main.js + booking.js functionality
         */

        // ===== BOOKING SYSTEM CONFIGURATION =====
        const BookingSystem = {
            CONFIG: {
                mode: 'localStorage', // 'localStorage' | 'webhook' | 'api' | 'calendly'
                webhookUrl: '',
                apiUrl: '',
                apiKey: '',
                thirdPartyEmbed: '',
                caféName: 'The Coffee House',
                phone: '0337 1666607',
                whatsapp: '923371666607',
                email: 'info@thecoffeehouse.pk',
                openingHours: {
                    weekday: { open: 8, close: 22 },
                    weekend: { open: 9, close: 23 }
                },
                slotDuration: 60,
                minAdvanceHours: 2,
                maxAdvanceDays: 30,
                maxPartySize: 6,
                timeSlots: [
                    '08:00', '09:00', '10:00', '11:00', '12:00', '13:00',
                    '14:00', '15:00', '16:00', '17:00', '18:00', '19:00',
                    '20:00', '21:00', '22:00'
                ]
            },
            state: { bookings: [], selectedDate: null, selectedTime: null },
            
            init() {
                if (document.getElementById('bookingForm')) {
                    this.setupForm();
                    this.populateTimeSlots();
                    this.setMinDate();
                    this.loadBookings();
                    this.attachEventListeners();
                }
            },
            
            setupForm() {
                const today = new Date().toISOString().split('T')[0];
                document.getElementById('bookingDate').min = today;
            },
            
            populateTimeSlots() {
                const timeSelect = document.getElementById('bookingTime');
                if (!timeSelect) return;
                timeSelect.innerHTML = '<option value="">Select time</option>';
                this.CONFIG.timeSlots.forEach(time => {
                    const option = document.createElement('option');
                    option.value = time;
                    option.textContent = this.formatTime(time);
                    timeSelect.appendChild(option);
                });
            },
            
            setMinDate() {
                const dateInput = document.getElementById('bookingDate');
                if (!dateInput) return;
                const now = new Date();
                now.setHours(now.getHours() + this.CONFIG.minAdvanceHours);
                dateInput.min = now.toISOString().split('T')[0];
            },
            
            attachEventListeners() {
                const form = document.getElementById('bookingForm');
                const dateInput = document.getElementById('bookingDate');
                
                form.addEventListener('submit', (e) => this.handleSubmit(e));
                dateInput.addEventListener('change', (e) => {
                    this.state.selectedDate = e.target.value;
                    this.updateAvailableTimes();
                });
                
                const phoneInput = document.getElementById('bookingPhone');
                phoneInput.addEventListener('input', (e) => {
                    let value = e.target.value.replace(/\D/g, '');
                    if (value.startsWith('0')) value = '92' + value.slice(1);
                    if (!value.startsWith('92')) value = '92' + value;
                    e.target.value = value.slice(0, 12);
                });
            },
            
            updateAvailableTimes() {
                const dateInput = document.getElementById('bookingDate');
                const timeSelect = document.getElementById('bookingTime');
                const selectedDate = new Date(dateInput.value);
                const now = new Date();
                
                const day = selectedDate.getDay();
                const isWeekend = (day === 0 || day === 6);
                const hours = isWeekend ? this.CONFIG.openingHours.weekend : this.CONFIG.openingHours.weekday;
                
                timeSelect.innerHTML = '<option value="">Select time</option>';
                
                this.CONFIG.timeSlots.forEach(time => {
                    const [hour] = time.split(':').map(Number);
                    if (hour < hours.open || hour >= hours.close) return;
                    
                    const today = new Date().toDateString();
                    if (selectedDate.toDateString() === today) {
                        const [selHour, selMin] = time.split(':').map(Number);
                        const slotTime = new Date();
                        slotTime.setHours(selHour, selMin, 0, 0);
                        const minTime = new Date(now.getTime() + this.CONFIG.minAdvanceHours * 60 * 60 * 1000);
                        if (slotTime < minTime) return;
                    }
                    
                    if (this.CONFIG.mode === 'localStorage' && this.isTimeBooked(dateInput.value, time)) return;
                    
                    const option = document.createElement('option');
                    option.value = time;
                    option.textContent = this.formatTime(time);
                    timeSelect.appendChild(option);
                });
            },
            
            isTimeBooked(date, time) {
                return this.state.bookings.some(booking => booking.date === date && booking.time === time);
            },
            
            async handleSubmit(e) {
                e.preventDefault();
                const formData = this.collectFormData();
                if (!this.validateForm(formData)) return;
                
                const submitBtn = document.getElementById('bookingSubmit');
                const originalText = submitBtn.innerHTML;
                submitBtn.disabled = true;
                submitBtn.innerHTML = '<i class="fas fa-spinner fa-spin"></i> Processing...';
                
                try {
                    switch (this.CONFIG.mode) {
                        case 'localStorage': await this.processLocalStorage(formData); break;
                        case 'webhook': await this.processWebhook(formData); break;
                        case 'api': await this.processAPI(formData); break;
                        case 'calendly': case 'opentable': this.redirectToThirdParty(formData); return;
                        default: await this.processLocalStorage(formData);
                    }
                    this.showSuccess(formData);
                } catch (error) {
                    console.error('Booking error:', error);
                    alert('Sorry, there was an error. Please try again or call us directly.');
                } finally {
                    submitBtn.disabled = false;
                    submitBtn.innerHTML = originalText;
                }
            },
            
            collectFormData() {
                return {
                    id: Date.now().toString(),
                    name: document.getElementById('bookingName').value.trim(),
                    phone: document.getElementById('bookingPhone').value.trim(),
                    email: document.getElementById('bookingEmail').value.trim(),
                    guests: document.getElementById('bookingGuests').value,
                    date: document.getElementById('bookingDate').value,
                    time: document.getElementById('bookingTime').value,
                    occasion: document.getElementById('bookingOccasion').value,
                    notes: document.getElementById('bookingNotes').value.trim(),
                    timestamp: new Date().toISOString(),
                    status: 'confirmed'
                };
            },
            
            validateForm(data) {
                if (!data.name || !data.phone || !data.guests || !data.date || !data.time) {
                    alert('Please fill in all required fields.');
                    return false;
                }
                const phoneRegex = /^92\d{10}$/;
                if (!phoneRegex.test(data.phone)) {
                    alert('Please enter a valid Pakistani phone number (e.g., 923371666607)');
                    return false;
                }
                const selectedDate = new Date(data.date);
                const now = new Date();
                if (selectedDate < now) {
                    alert('Please select a future date.');
                    return false;
                }
                return true;
            },
            
            async processLocalStorage(data) {
                const stored = localStorage.getItem('coffeeHouseBookings');
                this.state.bookings = stored ? JSON.parse(stored) : [];
                this.state.bookings.push(data);
                localStorage.setItem('coffeeHouseBookings', JSON.stringify(this.state.bookings));
                return new Promise(resolve => setTimeout(resolve, 800));
            },
            
            async processWebhook(data) {
                if (!this.CONFIG.webhookUrl) throw new Error('Webhook URL not configured');
                const response = await fetch(this.CONFIG.webhookUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ ...data, café: this.CONFIG.caféName, source: 'website' })
                });
                if (!response.ok) throw new Error('Webhook failed');
                return response.json();
            },
            
            async processAPI(data) {
                if (!this.CONFIG.apiUrl) throw new Error('API URL not configured');
                const headers = {
                    'Content-Type': 'application/json',
                    ...(this.CONFIG.apiKey && { 'Authorization': `Bearer ${this.CONFIG.apiKey}` })
                };
                const response = await fetch(this.CONFIG.apiUrl, {
                    method: 'POST', headers, body: JSON.stringify(data)
                });
                if (!response.ok) {
                    const error = await response.json().catch(() => ({}));
                    throw new Error(error.message || 'API request failed');
                }
                return response.json();
            },
            
            redirectToThirdParty(data) {
                if (this.CONFIG.thirdPartyEmbed) {
                    const url = new URL(this.CONFIG.thirdPartyEmbed);
                    url.searchParams.append('name', data.name);
                    url.searchParams.append('email', data.email);
                    url.searchParams.append('phone', data.phone);
                    window.location.href = url.toString();
                }
            },
            
            showSuccess(data) {
                document.querySelector('.booking-form').classList.add('hidden');
                const successDiv = document.getElementById('bookingSuccess');
                successDiv.classList.remove('hidden');
                document.getElementById('confirmedName').textContent = data.name;
                document.getElementById('confirmedGuests').textContent = data.guests;
                document.getElementById('confirmedDate').textContent = this.formatDate(data.date);
                document.getElementById('confirmedTime').textContent = this.formatTime(data.time);
                document.getElementById('confirmedContact').textContent = data.email || data.phone;
                successDiv.scrollIntoView({ behavior: 'smooth', block: 'center' });
            },
            
            formatTime(time24) {
                const [hour, minute] = time24.split(':');
                const h = parseInt(hour);
                const ampm = h >= 12 ? 'PM' : 'AM';
                const h12 = h % 12 || 12;
                return `${h12}:${minute} ${ampm}`;
            },
            
            formatDate(dateStr) {
                const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
                return new Date(dateStr).toLocaleDateString('en-PK', options);
            },
            
            loadBookings() {
                if (this.CONFIG.mode === 'localStorage') {
                    const stored = localStorage.getItem('coffeeHouseBookings');
                    this.state.bookings = stored ? JSON.parse(stored) : [];
                }
            }
        };

        function initNavbar() {
            const navbar = document.getElementById('navbar');
            const mobileMenuBtn = document.getElementById('mobileMenuBtn');
            const navLinks = document.getElementById('navLinks');
            
            window.addEventListener('scroll', function() {
                if (window.scrollY > 50) navbar.classList.add('scrolled');
                else navbar.classList.remove('scrolled');
            });
            
            if (mobileMenuBtn) {
                mobileMenuBtn.addEventListener('click', function() {
                    navLinks.classList.toggle('active');
                    const icon = this.querySelector('i');
                    if (navLinks.classList.contains('active')) {
                        icon.classList.remove('fa-bars');
                        icon.classList.add('fa-times');
                    } else {
                        icon.classList.remove('fa-times');
                        icon.classList.add('fa-bars');
                    }
                });
            }
            
            document.querySelectorAll('.nav-links a').forEach(link => {
                link.addEventListener('click', () => {
                    navLinks.classList.remove('active');
                    const icon = mobileMenuBtn?.querySelector('i');
                    if (icon) {
                        icon.classList.remove('fa-times');
                        icon.classList.add('fa-bars');
                    }
                });
            });
        }

        function initScrollAnimations() {
            const observerOptions = { threshold: 0.1, rootMargin: '0px 0px -50px 0px' };
            const observer = new IntersectionObserver(function(entries) {
                entries.forEach(entry => {
                    if (entry.isIntersecting) entry.target.classList.add('visible');
                });
            }, observerOptions);
            document.querySelectorAll('.fade-in').forEach(el => observer.observe(el));
        }

        function initMenuFilter() {
            const tabs = document.querySelectorAll('.menu-tab');
            const items = document.querySelectorAll('.menu-item');
            tabs.forEach(tab => {
                tab.addEventListener('click', function() {
                    tabs.forEach(t => t.classList.remove('active'));
                    this.classList.add('active');
                    const category = this.dataset.category;
                    items.forEach(item => {
                        if (category === 'all' || item.dataset.category === category) {
                            item.classList.remove('hidden');
                            setTimeout(() => { item.style.opacity = '1'; item.style.transform = 'translateY(0)'; }, 10);
                        } else {
                            item.style.opacity = '0';
                            item.style.transform = 'translateY(20px)';
                            setTimeout(() => { item.classList.add('hidden'); }, 300);
                        }
                    });
                });
            });
        }

        function initSmoothScroll() {
            document.querySelectorAll('a[href^="#"]').forEach(anchor => {
                anchor.addEventListener('click', function(e) {
                    e.preventDefault();
                    const target = document.querySelector(this.getAttribute('href'));
                    if (target) {
                        const offsetTop = target.offsetTop - 80;
                        window.scrollTo({ top: offsetTop, behavior: 'smooth' });
                    }
                });
            });
        }
